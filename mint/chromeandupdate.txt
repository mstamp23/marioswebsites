wget -qO /tmp/chrome.deb https://dl.google.com/linux/direct/google-chrome-stable_current_amd64.deb && sudo apt install -y /tmp/chrome.deb


#!/bin/bash
sudo apt update
sudo apt full-upgrade -y
sudo apt autoremove -y
sudo apt autoclean
read -rp "Press Enter to close..." < /dev/tty

chmod +x /home/m/update.sh

sudo visudo
m ALL=(ALL) NOPASSWD: /usr/bin/apt

cat > ~/Desktop/update.desktop << 'EOF'
[Desktop Entry]
Version=1.0
Name=System Update
Comment=Run system update script
Exec=xfce4-terminal -e "bash -c '/home/m/update.sh'"
Icon=system-software-update
Terminal=false
Type=Application
Categories=System;
EOF
chmod +x ~/Desktop/update.desktop


sudo add-apt-repository ppa:phoerious/keepassxc
sudo apt update
sudo apt install keepassxc
